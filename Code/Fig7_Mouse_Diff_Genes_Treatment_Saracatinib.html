for inst_col in df_gene_pval:
    
    inst_pvals = df_gene_pval[inst_col].sort_values()
    inst_genes = inst_pvals.index.tolist()
    rej, pval_corr = smm.multipletests(inst_pvals, 0.05, method='fdr_bh')[:2]
    
    # sort all series
    rows = sorted(df_sig.index.tolist())
    
    ser_mean = all_fold_info[inst_col]['cluster_mean'][rows]
    ser_mean.name = 'Mean'
    ser_other = all_fold_info[inst_col]['other_mean'][rows]
    ser_other.name = 'Other Mean'
    ser_fold = all_fold_info[inst_col]['log2_fold'][rows]
    ser_fold.name = 'Log2 Fold Change'
    
    ser_pval = pd.Series(data=inst_pvals, index=inst_genes, name='P-values')[rows]
    ser_bh_pval = pd.Series(data=pval_corr, index=inst_genes, name='BH P-values')[rows]
    
    list_ser = [ser_mean, ser_other, ser_fold, ser_pval, ser_bh_pval]
    inst_df = pd.concat(list_ser, axis=1)
    
    # sort based on bh pval
    rows_bh = inst_df['P-values'].sort_values().index.tolist()
    inst_df = inst_df.loc[rows_bh]
    
    df[inst_col] = inst_df
    
    inst_df.to_csv('../data/processed/' + set_name + '/' + inst_col + '.csv')
